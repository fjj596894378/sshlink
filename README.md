# sshlink
运行该工程，可以直接连接外网数据库


-----------------------------------------------
运行，配置filepath文件
-----------------------------------------------

#本地的端口
localport=1522
#数据库所在的服务器ip地址
remotehost=192.168.1.168  
#数据库所在的服务器端口
remoteport=1521
#隧道机的用户名
userTunnel=ABC
#隧道机的密码
passwordTunnel=ABCPassword
#隧道机的服务器ip地址
hostTunnel=10.10.10.10
#隧道机开放的端口
portTunnel=22
